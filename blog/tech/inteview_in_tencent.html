题目：

小 Q 得到了一个长度为 n 的序列 A，A 中的数各不相同。对于 A 中的每一个数 ， 求：

以及令上式取得最小值的 （记为 ）。若最小值点不唯一，则选择使 较小的那个。

question_5 <- function(n, v){
  diff <- NULL
  combination <- gtools::combinations(n, 2)
  for(i in 1:length(combination[,1])){
    diff[i] <- v[combination[i,]][1] - v[combination[i,]][2]
    if (v[combination[i,]][1] == v[combination[i,]][2]){
      break
    }
    if (v[combination[i,]][1] >= v[combination[i,]][2] & diff[i] != 0){
      cat(abs(diff[i]), " ", v[combination[i,]][2], "\n")
    }
    if (v[combination[i,]][1] < v[combination[i,]][2] & diff[i] != 0){
      cat(abs(diff[i]), " ", v[combination[i,]][1], "\n")
    }
  }
}
